def normalise_dataset(df):
    X = df.drop(columns=['Dataset'])
    y = df['Dataset']

    scaler = MinMaxScaler()

    X_normalized = scaler.fit_transform(X)

    df_normalized = pd.DataFrame(X_normalized, columns=X.columns)

    df_normalized['Target'] = y
    
    return df_normalized

def split_data(dataframe, test_size):
    # splitting data into train and test

    X = dataframe.drop(['Dataset'], axis=1)
    y = dataframe['Dataset']

    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=test_size, random_state=42)
    
    print("Results for test size: ", test_size*100, 'percent')
    return X, y, X_train, X_test, y_train, y_test
  • With a single train-test split, the model's performance may be highly dependent on the particular random partitioning of the data. K-fold cross-validation mitigates this by averaging the model's performance over multiple train-test splits. This provides a more robust and reliable estimate of how well the model generalizes to unseen data.
In [21]:
def k_fold_validation(X, y):
    k = 5

    # Create the k-fold cross-validation object
    kf = KFold(n_splits=k, shuffle=True, random_state=42)

    model = RandomForestClassifier()

    scores = cross_val_score(model, X, y, cv=kf)

    print("Cross-validation scores for RandomForest:", scores)
    print("Mean CV score for RandomForest:", scores.mean())
    
    kf = KFold(n_splits=k, shuffle=True, random_state=42)
    model = KNeighborsClassifier()

    scores = cross_val_score(model, X, y, cv=kf)

    print("Cross-validation scores for KNN:", scores)
    print("Mean CV score for KNN:", scores.mean())

def train_model_and_results_for_random_forrest(X_train, y_train, X_test, y_test):
    
    param_grid = {
        'n_estimators': [10, 20, 30, 40, 50],     # Number of trees in the forest
        'max_depth': [None, 1, 2, 3, 4, 5],       # Maximum depth of each tree
        'min_samples_split': [2, 5, 10],       # Minimum samples required to split an internal node
        'min_samples_leaf': [1, 2, 4],         # Minimum samples required to be at a leaf node
        'max_features': ['auto', 'sqrt', 'log2']  # Number of features to consider at each split
    }

    rf_classifier = RandomForestClassifier(random_state=42)

    grid_search = GridSearchCV(estimator=rf_classifier, param_grid=param_grid, cv=5, scoring='accuracy')
    grid_search.fit(X_train, y_train)

    best_params = grid_search.best_params_

    best_rf_classifier = RandomForestClassifier(random_state=42, **best_params)
    best_rf_classifier.fit(X_train, y_train)

    rf_predictions = best_rf_classifier.predict(X_test)

    rf_accuracy = accuracy_score(y_test, rf_predictions)

    print("Best Hyperparameters for Random Forrest:", best_params)
    print("Random Forest Accuracy with Best Hyperparameters:", rf_accuracy)

def train_model_and_results_for_KNN(X_train, y_train, X_test, y_test):
    
    k_values = np.arange(1, 11)

    accuracy_scores = []

    for k in k_values:
        knn = KNeighborsClassifier(n_neighbors=k)
        knn.fit(X_train, y_train)
        y_pred = knn.predict(X_test)
        accuracy = accuracy_score(y_test, y_pred)
        accuracy_scores.append(accuracy)

    plt.figure(figsize=(8, 6))
    plt.plot(k_values, accuracy_scores, marker='o', linestyle='-', color='b')
    plt.title('Elbow Method for Optimal K')
    plt.xlabel('Number of Neighbors (K)')
    plt.ylabel('Accuracy')
    plt.grid(True)
    plt.show()

  • Metrics Chosen

    • Accuracy: Accuracy is the most straightforward metric and is often used as a general performance measure. It calculates the ratio of correctly predicted instances to the total number of instances. However, accuracy may not be suitable for imbalanced datasets.

    • Precision and Recall: These metrics are important when dealing with imbalanced datasets. Precision measures the ratio of correctly predicted positive instances to the total predicted positive instances. Recall (Sensitivity or True Positive Rate) measures the ratio of correctly predicted positive instances to the total actual positive instances

    • Area Under the Receiver Operating Characteristic Curve (AUC-ROC): ROC curves plot the True Positive Rate (TPR) against the False Positive Rate (FPR) at various threshold settings. AUC-ROC measures the area under the ROC curve and is used to evaluate binary classification models. It quantifies the model's ability to distinguish between positive and negative classes.

Performance Evaluation¶

  • Random Forest has a high recall of 1.0, which means it is excellent at correctly identifying positive instances. However, this comes at the cost of lower precision (0.7436), indicating that it may produce more false positives. The ROC AUC score of 0.7983 suggests that the classifier's ability to discriminate between the two classes is reasonably good.

  • KNN achieves a higher accuracy (0.7607) compared to Random Forest, indicating that it has a higher overall correct prediction rate. Additionally, KNN shows a better balance between precision (0.8471) and recall (0.8276) compared to Random Forest. The ROC AUC score (0.7828) suggests good discrimination capability, though slightly lower than Random Fores

  • We have to prioritize high recall as we should be more concerned about capturing as many true positives as possible, Random Forest might be a suitable choice due to its perfect recall (1.0).
